\section{Selected Publications}

\cvitem{\textbf{Subhransu S.\ Bhattacharjee}, Dylan Campbell \& Rahul Shome}{\textit{Believing is Seeing: Unobserved Object Detection using Generative Models}. \textit{IEEE/CVF Conference on Computer Vision and Pattern Recognition (CVPR)}, 2025. \href{https://1ssb.github.io/UOD/}{Project page}.
\\ \textit{Uses pre-trained generative models to infer spatio-semantic distributions of occluded and out-of-frame objects across 2D, 2.5D, and 3D scenes.}
}

\cvitem{\textbf{Subhransu S.\ Bhattacharjee}, Dylan Campbell \& Rahul Shome}{\textit{FlatLands: Generative Floormap Completion From a Single Egocentric View}. \textit{European Conference on Computer Vision (ECCV)}, 2026. Springer. \href{https://arxiv.org/abs/2603.16016}{Paper}.
\\ \textit{Releases FlatLands: 270,575 observations from 17,656 real metric indoor scenes across six datasets.}
\\ \textit{Benchmarks single-view BEV floor completion with aligned observation, visibility, validity, and ground-truth maps.}
}

\cvitem{\textbf{Subhransu S.\ Bhattacharjee}, Hao Lu, Dylan Campbell \& Rahul Shome}{\textit{MatterDoor: Sampling Zero-shot Spatio\mbox{-}semantic Priors using Generative Models}. \textit{arXiv:2510.11014}, 2025. Under review. \href{https://arxiv.org/pdf/2510.11014}{Preprint}.
\\
\textit{Samples environment priors from partial observations using pretrained generative models.}
\\
\textit{Produces RGB-D point-cloud hypotheses for occupancy, target semantics, and configuration-space planning.}
}

\cvitem{\textbf{Subhransu S.\ Bhattacharjee} \& Ian Petersen}{\textit{Analysis of the Whiplash Gradient Descent Dynamics}. \textit{Asian Journal of Control} (Special Issue), Wiley, 2023. DOI: \href{https://onlinelibrary.wiley.com/doi/10.1002/asjc.3153}{10.1002/asjc.3153}.
\\
\textit{A feedback-based inertial gradient method with adaptive damping achieves provable polynomial and exponential convergence rates that extend classical optimisation theory to high-dimensional settings.}
}
